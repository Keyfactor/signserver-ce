    Installation

    Add user for running the application server

    $ sudo adduser jboss

    Use password "foo123".

    Install dependencies

    $ sudo apt install unzip
    $ sudo apt install mariadb-server mariadb-client
    $ sudo apt install openjdk-8-jdk ant
    $ sudo apt install ntp ntpdate ntp-doc
    $ sudo apt install apache2
    $ sudo apt install apg
    $ sudo apt install python3

    Install software files

    $ sudo mkdir /opt/install_disk
    $ cd
    $ sudo mv jboss-eap-6.4-signserver-3.zip* /opt/install_disk/
    $ sudo cp signserver-ee-4.1.0-pk2-bin.zip
    ...

    Configure NTP

    Edit /etc/ntp.conf and uncomment all loop and add:

    server hive.primekey.se

    Install JBoss (Quick-Start) files

    $ cd
    $ unzip /opt/install_disk/jboss-eap-6.4-signserver-3.zip
    $ sudo mv jboss-eap-6.4-signserver /opt/
    $ sudo chown jboss:jboss -R /opt/jboss-eap-6.4-signserver/

    Install SignServer files

    $ cd
    $ unzip /opt/install_disk/signserver-ee-4.1.0-pk3-bin.zip
    $ sudo mv signserver-ee-4.1.0-pk3 /opt/
    $ sudo chown jboss:jboss -R /opt/signserver-ee-4.1.0-pk3/
    $ sudo ln -s signserver-ee-4.1.0-pk3 signserver

    Configure MariaDB

    Below [mysqld] in /etc/mysql/mariadb.conf.d/50-server.cnf:

    binlog_format=row

    Create database

    sudo mysql -uroot -p
    create database signserver;
    grant all on signserver.* to signserver@localhost identified by 'foo123';

    Configure SignServer

    $ sudo su - jboss
    $ cd /opt/signserver-ee-4.1.0-pk3/
    $ cp conf/databaseprotection.properties.sample conf/databaseprotection.properties
    $ cp conf/signserver_deploy.properties.sample conf/signserver_deploy.properties
    $ cp /opt/install_disk/demo-workers.properties conf/zz-RESET-DEMO-CONFIG.properties
    $ cp conf/timemonitor.properties.sample conf/timemonitor.properties

    Edit conf/databaseprotection.properties and set:

    databaseprotection.enablesign.AuditRecordData = true
    databaseprotection.enableverify.AuditRecordData = true

    Edit conf/signserver_deploy.properties and set:

    database.name=mysql
    httpserver.external.privhttps=443
    web.admingui.dist.enabled=true
    web.clientcli.dist.enabled=true
    healthcheck.authorizedips=ANY
    statusrepository.log=NONE
    dist.includes.extra=p12/trusted-cas.pem
    dist.sampleconfigs.includes.extra=conf/zz-RESET-DEMO-CONFIG.properties

    Edit conf/admingui_default.properties and change to /set:

    url=https://dss-demo1.primekey.se/frymcemtus7/signserver
    protocol=WS
    $ mkdir p12
    $ cp /opt/acme-tiny/keys/intermediate.pem p12/trusted-cas.pem

    Edit conf/timemonitor.properties

    signserver.process.url=http://localhost:8080/frymcemtus7/signserver/process

    Edit /home/jboss/.bashrc and add:

    export APPSRV_HOME=/opt/jboss-eap-6.4-signserver

    Deploy SignServer

    $ source ~/.bashrc
    $ bin/ant deploy

    (Deploy again as workaround for DSS-xyz)

    $ bin/ant deploy

    Configure Apache HTTP Server

    $ sudo cp /etc/apache2/sites-available/000-default.conf /etc/apache2/sites-available/dss-demo1.conf

    Edit dss-demo1.conf:

    ServerName dss-demo1
    ServerAdmin sysop@primekey.se
    DocumentRoot /opt/www/html
    ErrorLog ${APACHE_LOG_DIR}/dss-demo1-error.log
    CustomLog ${APACHE_LOG_DIR}/dss-demo1-access.log combined
    <Directory "/opt/www/html/">
    Require all granted
    </Directory>
    ProxyRequests Off
    <Proxy balancer://mycluster-3>
    BalancerMember ajp://localhost:8009/frymcemtus7/signserver
    </Proxy>
    ProxyPass /frymcemtus7/signserver balancer://mycluster-3
    RewriteEngine On

    Enable site

    $ sudo a2ensite dss-demo1
    $ sudo a2enmod proxy_ajp
    $ sudo a2enmod rewrite
    $ sudo a2enmod proxy_balancer
    $ sudo a2enmod lbmethod_byrequests

    Configure info-page

    $ sudo mkdir /opt/www
    $ sudo mkdir /opt/www/html
    $ sudo cp /var/www/html/index.html /opt/www/html/
    $ sudo chmod a+r -R /var/www

    Modify /opt/www/index.html:
    - title: SignServer Demo Server 1
    - class="page_header...":
    SignServer Demo Server 1
    Remove img ubuntu logo

    Add to standalone.xml:

    <connector name="ajp" protocol="AJP/1.3" scheme="http" socket-binding="ajp"/>

    Enable Firewall

    $ sudo ufw allow 22/tcp
    $ sudo ufw enable
    $ sudo ufw allow 80/tcp
    $ sudo ufw allow 443/tcp
    $ sudo mkdir /opt/www/html/frymcemtus7
    $ sudo cp /opt/www/html/index.html /opt/www/html/frymcemtus7/
    $ sudo cp /opt/signserver-ee-4.1.0-pk3/res/test/dss10/dss10_admin1.p12 /opt/www/html/frymcemtus7/
    $ sudo cp /opt/signserver-ee-4.1.0-pk3/res/test/dss10/dss10_admin1.p12 /opt/www/html/frymcemtus7/
    $ sudo cp /opt/signserver-ee-4.1.0-pk3/res/test/dss10/DSSRootCA10.cacert.pem /opt/www/html/frymcemtus7/dss10_root.crt
    $ sudo chmod a+r -R /var/www

    Set up Lets Encrypt

    $ cd
    $ git clone https://github.com/diafygi/acme-tiny.git acme-tiny-git
    Read the entire script.
    $ sudo mv acme-tiny-git /opt/
    $ cd /opt
    $ sudo ln -s acme-tiny-git acme-tiny
    $ mkdir keys
    $ openssl genrsa 4096 > account.key
    $ openssl genrsa 2048 > dss-demo1.key
    $ openssl req -new -sha256 -key dss-demo1.key -subj "/" -reqexts SAN -config <(cat /etc/ssl/openssl.cnf <(printf "[SAN]\nsubjectAltName=DNS:dss-demo1.primekey.se,DNS=dss-demo1.primekey.com")) > dss-demo1.csr

    Set up ACME challenges

    $ sudo mkdir /opt/www/challenges
    $ sudo chmod a+w /opt/www/challenges/
    $ sudo chown www-data:www-data /opt/www/challenges

    Edit /etc/apache2/sites-available/dss-demo1.conf and add:

    Alias /.well-known/acme-challenge /opt/www/challenges
    <Directory "/opt/www/html/">
    Require all granted
    </Directory>
    $ sudo service apache2 reload

    Get the certificates

    $ cd /opt/acme-tiny
    $ python3 acme_tiny.py --account-key ./keys/account.key --csr ./keys/dss-demo1.csr --acme-dir /opt/www/challenges/ > ./keys/dss-demo1-2.crt
    $ wget -O - https://letsencrypt.org/certs/lets-encrypt-x3-cross-signed.pem > ./keys/intermediate.pem

    Configure the virtual host

    $ sudo cp /etc/apache2/sites-available/dss-demo1.conf /etc/apache2/sites-available/dss-demo1-ssl.conf

    Edit dss-demo1-ssl.conf to add TLS:

    <VirtualHost *:443>
    ...
    SSLCertificateFile /opt/acme-tiny/keys/dss-demo1.crt
    SSLCertificateKeyFile /opt/acme-tiny/keys/dss-demo1.key
    SSLCertificateChainFile /opt/acme-tiny/keys/intermediate.pem
    SSLProtocol all -SSLv2 -SSLv3
    SSLCipherSuite ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-AES256-GCM-SHA384:DHE-RSA-AES128-GCM-SHA256:DHE-DSS-AES128-GCM-SHA256:kEDH+AESGCM:ECDHE-RSA-AES128-SHA256:ECDHE-ECDSA-AES128-SHA256:ECDHE-RSA-AES128-SHA:ECDHE-ECDSA-AES128-SHA:ECDHE-RSA-AES256-SHA384:ECDHE-ECDSA-AES256-SHA384:ECDHE-RSA-AES256-SHA:ECDHE-ECDSA-AES256-SHA:DHE-RSA-AES128-SHA256:DHE-RSA-AES128-SHA:DHE-DSS-AES128-SHA256:DHE-RSA-AES256-SHA256:DHE-DSS-AES256-SHA:DHE-RSA-AES256-SHA:AES128-GCM-SHA256:AES256-GCM-SHA384:AES128-SHA256:AES256-SHA256:AES128-SHA:AES256-SHA:AES:CAMELLIA:DES-CBC3-SHA:!aNULL:!eNULL:!EXPORT:!DES:!RC4:!MD5:!PSK:!aECDH:!EDH-DSS-DES-CBC3-SHA:!EDH-RSA-DES-CBC3-SHA:!KRB5-DES-CBC3-SHA
    SSLHonorCipherOrder on
    SSLCompression off
    SSLOptions +StrictRequire
    $ sudo a2ensite dss-demo1-ssl.conf
    $ sudo a2enmod ssl
    $ sudo service apache2 restart

    Test with SSLLabs:

    https://www.ssllabs.com/ssltest/analyze.html?d=dss%2ddemo1.primekey.se&hideResults=on&latest
    Result: "A"

    Setup optional client certificate authentication

    $ wget https://download.primekey.se/keys/PrimeCA1-pem.crt
    $ sudo mv PrimeCA1-pem.crt /opt/www/html/frymcemtus7/
    $ sudo su -
    # cat /opt/www/html/frymcemtus7/dss10_root.crt /opt/www/html/frymcemtus7/PrimeCA1-pem.crt > /opt/www/html/frymcemtus7/trust-anchors.pem
    # exit

    Edit dss-demo1-ssl.conf and add:

    # Optional client certificate authentication
    SSLVerifyClient optional
    SSLVerifyDepth 3
    SSLCACertificateFile /opt/www/html/frymcemtus7/trust-anchors.pem
    # Require for AdminWS and AdminWeb
    <Location /frymcemtus7/signserver/AdminWSService>
    SSLVerifyClient require
    </Location>
    <Location /frymcemtus7/signserver/adminweb>
    SSLVerifyClient require
    </Location>
    $ sudo service apache2 restart

    Install HSM emulator

    TODO: We probably want to use SoftHSM2 but I don't have time looking into that right now so will use SafeNet emulator instead

    $ cd
    $ tar zxvf /opt/install_disk/ETcpsdk.tar.gz
    $ sudo mv ETcpsdk/ /opt/

    Setup HSM emulator

    $ sudo nano /etc/environment
    LD_LIBRARY_PATH=/opt/ETcpsdk/lib/linux-x86_64
    ET_PTKC_SW_AUTOSEEDRNG=true
    $ sudo su - jboss

    Use password foo123 for all HSM operations.

    $ /opt/ETcpsdk/bin/linux-x86_64/ctconf
    $ /opt/ETcpsdk/bin/linux-x86_64/ctconf -c10
    $ /opt/ETcpsdk/bin/linux-x86_64/ctconf -fc
    $ /opt/ETcpsdk/bin/linux-x86_64/ctkmu t -s1 -lmyslot1
    $ /opt/ETcpsdk/bin/linux-x86_64/ctkmu t -s2 -lmyslot2
    $ /opt/ETcpsdk/bin/linux-x86_64/ctkmu t -s3 -lmyslot3
    $ /opt/ETcpsdk/bin/linux-x86_64/ctkmu t -s4 -lmyslot4
    $ /opt/ETcpsdk/bin/linux-x86_64/ctkmu t -s5 -lmyslot5

    Restart SignServer (JBoss)

    Setup workers

    In AdminWeb choose:
    Workers -> Add... -> From Template -> RESET-DEMO-CONFIG.properties

    Start SignServer TimeMonitor

    $ bin/timemonitor-in-background.sh

    Setup init.d scripts

    $ sudo cp /opt/install_disk/init.d-timemonitor /etc/init.d/timemonitor
    $ sudo chmod +x /etc/init.d/timemonitor
    $ sudo update-rc.d timemonitor defaults
    $ sudo cp /opt/install_disk/init.d-jboss /etc/init.d/jboss
    $ sudo chmod +x /etc/init.d/jboss
    $ sudo update-rc.d jboss defaults

    Certificate Renewal

    cd /opt/acme-tiny
    python3 acme_tiny.py --account-key ./keys/account.key --csr ./keys/dss-demo1-2.csr --acme-dir /opt/www/challenges/ > ./keys/dss-demo1-2.crt
    sudo service apache2 restart
